    ISP (Interface Segregation Principle, ВЮИьё░ьјўВЮ┤Віц вХёвдг ВЏљВ╣Ў) "ьЂ┤вЮ╝ВЮ┤ВќИьіИвіћ ВъљВІаВЮ┤ ВѓгВџЕьЋўВДђ ВЋівіћ ВЮИьё░ьјўВЮ┤ВіцВЌљ ВЮўВА┤ьЋўВДђ ВЋіВЋёВЋ╝ ьЋювІц." вџ▒вџ▒ьЋю ВЮИьё░ьјўВЮ┤Віц ьЋўвѓўв│┤вІц ВъЉЖ│а ЖхгВ▓┤ВаЂВЮИ ВЮИьё░ьјўВЮ┤Віц ВЌгвЪг Ж░юЖ░ђ вѓФвІц. PythonВЌљВёювіћ ABC(Abstract Base Class)вѓў ProtocolВЮё ВѓгВџЕьЋ┤Вёю ВЮИьё░ьјўВЮ┤ВіцвЦ╝ ВаЋВЮўьЋювІц. (JavaВЌљВёювіћ interface ьѓцВЏївЊювАю, GoВЌљВёювіћ implicit interfaceвАю Ж░ЎВЮђ ВЏљВ╣ЎВЮё ВаЂВџЕьЋювІц.)

    D - DIP (Dependency Inversion Principle, ВЮўВА┤Вё▒ ВЌГВаё ВЏљВ╣Ў)¶

    "ВЃЂВюё вфевЊѕВЮ┤ ьЋўВюё вфевЊѕВЌљ ВЮўВА┤ьЋўвЕ┤ ВЋѕ вљювІц. вЉў вІц ВХћВЃЂВЌљ ВЮўВА┤ьЋ┤ВЋ╝ ьЋювІц."

    ВЮ┤Ж▓ї SRP вІцВЮїВю╝вАю ВцЉВџћьЋю ВЏљВ╣ЎВЮ┤вІц. ВЮ┤ ВЏљВ╣ЎВЌљВёю DI(Dependency Injection)ВЎђ IoC(Inversion of Control)Ж░ђ ьїїВЃЮвљювІц.

    DIP Вюёв░ў (ВЃЂВюёЖ░ђ ьЋўВюёВЌљ ВДЂВаЉ ВЮўВА┤):

    class OrderService:
    def __init__(self):
        # ВДЂВаЉ ВЃЮВё▒ьЋювІц. OrderServiceЖ░ђ MySQLRepositoryвЦ╝ "ВЋѕвІц"
        self.repo = MySQLOrderRepository()
        self.payment = KakaoPayGateway()

    ВЮ┤ ВйћвЊюВЮў вгИВаю: OrderServiceЖ░ђ MySQLOrderRepositoryвЦ╝ ВДЂВаЉ вДївЊавІц. MySQLВЮё PostgreSQLвАю в░ћЖЙИвацвЕ┤? OrderService ВйћвЊювЦ╝ ВѕўВаЋьЋ┤ВЋ╝ ьЋювІц. Ж▓░ВаювЦ╝ В╣┤В╣┤ВўцьјўВЮ┤ВЌљВёю ьєаВіцьјўВЮ┤вАю в░ћЖЙИвацвЕ┤? ВЌГВІю OrderServiceвЦ╝ ВѕўВаЋьЋ┤ВЋ╝ ьЋювІц.

    "ВБ╝вгИ Вёюв╣ёВіц"Ж░ђ "MySQL", "В╣┤В╣┤ВўцьјўВЮ┤", "ВЮ┤вЕћВЮ╝"ВЮ┤вЮ╝віћ ЖхгВ▓┤ВаЂВЮИ ЖхгьўёВЌљ вгХВЌг ВъѕвІц. ВЃЂВюё вфевЊѕ(OrderService)ВЮ┤ ьЋўВюё вфевЊѕ(MySQL, В╣┤В╣┤ВўцьјўВЮ┤)ВЌљ ВЮўВА┤ьЋўЖ│а Въѕвіћ Ж▒░вІц.

    DIP ВцђВѕў (ВХћВЃЂВЌљ ВЮўВА┤):

    class OrderRepository(ABC):
    """ВХћВЃЂ - ВЮИьё░ьјўВЮ┤ВіцвДї ВаЋВЮў"""
    @abstractmethod
    def save(self, order): ...

    @abstractmethod
    def find_by_id(self, order_id): ...


class MySQLOrderRepository(OrderRepository):
    """Жхгьўё - MySQL ВаёВџЕ"""
    def save(self, order):
        self.session.add(order)
        self.session.commit()

    def find_by_id(self, order_id):
        return self.session.query(Order).filter(Order.id == order_id).first()


class OrderService:
    def __init__(self, repo: OrderRepository):
        # ВХћВЃЂ ьЃђВъЁвДї ВЋїЖ│а ВъѕвІц.
        # MySQLВЮИВДђ PostgreSQLВЮИВДђ вфевЦИвІц. ВЋї ьЋёВџћвЈё ВЌєвІц.

    OrderServiceвіћ OrderRepository(ВХћВЃЂ)ВЌљвДї ВЮўВА┤ьЋювІц. MySQLВЮИВДђ PostgreSQLВЮИВДђ вфевЦИвІц. вѓўВцЉВЌљ PostgreSQLвАю в░ћЖЙИвацвЕ┤? PostgreSQLOrderRepositoryвЦ╝ вДївЊцЖ│а, OrderServiceвЦ╝ ВЃЮВё▒ьЋа вЋї ЖиИЖ▒И вёБВќ┤ВБ╝вЕ┤ вљювІц. OrderService ВйћвЊювіћ ьЋю ВцёвЈё ВЋѕ в░ћвђљвІц.

    ВЮ┤ "вёБВќ┤ВцђвІц"віћ ьќЅВюёЖ░ђ в░ћвАю DI(Dependency Injection, ВЮўВА┤Вё▒ ВБ╝ВъЁ)вІц.

    DIP (Dependency Inversion Principle, ВЮўВА┤Вё▒ ВЌГВаё ВЏљВ╣Ў) "ВЃЂВюё вфевЊѕВЮ┤ ьЋўВюё вфевЊѕВЌљ ВЮўВА┤ьЋўвЕ┤ ВЋѕ вљювІц. вЉў вІц ВХћВЃЂВЌљ ВЮўВА┤ьЋ┤ВЋ╝ ьЋювІц." SOLIDВЮў вДѕВДђвДЅВЮ┤Въљ SRP вІцВЮїВю╝вАю ВІцвг┤ В▓┤Ж░љВЮ┤ ьЂ░ ВЏљВ╣ЎВЮ┤вІц. ЖхгВ▓┤ВаЂВЮИ Жхгьўё(MySQL, Redis, В╣┤В╣┤ВўцьјўВЮ┤)ВЌљ ВДЂВаЉ ВЮўВА┤ьЋўвЕ┤, ЖиИ ЖхгьўёВЮё в░ћЖ┐ђ вЋї ВЃЂВюё вАюВДЂвЈё ВѕўВаЋьЋ┤ВЋ╝ ьЋювІц. ВХћВЃЂ(ВЮИьё░ьјўВЮ┤Віц)ВЌљ ВЮўВА┤ьЋўвЕ┤, ЖхгьўёвДї ЖхљВ▓┤ьЋўвЕ┤ вљювІц. (PythonВЌљВёювіћ ABCвѓў ProtocolвАю ВХћВЃЂВЮё ВаЋВЮўьЋювІц. JavaВЌљВёювіћ interface, GoВЌљВёювіћ implicit interface.)

    DIВЎђ IoC¶

    DI (Dependency Injection, ВЮўВА┤Вё▒ ВБ╝ВъЁ)¶

    ВЮўВА┤Вё▒ВЮё ьЂ┤въўВіц вѓ┤вХђВЌљВёю ВДЂВаЉ вДївЊцВДђ ВЋіЖ│а, ВЎИвХђВЌљВёю вёБВќ┤ВБ╝віћ ьїеьё┤ВЮ┤вІц.

    # DI ВЌєВЮї - ВДЂВаЉ ВЃЮВё▒
class OrderService:
    def __init__(self):
        self.repo = MySQLOrderRepository()  # ВДЂВаЉ вДївЊавІц

# DI ВъѕВЮї - ВЎИвХђВЌљВёю ВБ╝ВъЁ
class OrderService:
    def __init__(self, repo: OrderRepository):  # в░ќВЌљВёю в░ЏвіћвІц
        self.repo = repo

    ВЎИвХђВЌљВёю в░ЏВю╝вЕ┤ вГљЖ░ђ ВбІВЮђЖ░ђ?

    1. ьЁїВіцьіИьЋа вЋї Ж░ђВДю Ж░ЮВ▓┤(Mock)вЦ╝ вёБВЮё Вѕў ВъѕвІц. (Ch.21ВЌљВёю ВъљВёИьъѕ вІцвБгвІц.)
    2. MySQLВЮё PostgreSQLвАю ЖхљВ▓┤ьЋа вЋї OrderServiceвЦ╝ ВѕўВаЋьЋўВДђ ВЋіВЋёвЈё вљювІц.
    3. ВЮўВА┤Вё▒ВЮ┤ вфЁВІюВаЂВю╝вАю в│┤ВЮИвІц. __init__вДї в│┤вЕ┤ ВЮ┤ ьЂ┤въўВіцЖ░ђ вГў ьЋёВџћвАю ьЋўвіћВДђ ВЋї Вѕў ВъѕвІц.
    DI (Dependency Injection, ВЮўВА┤Вё▒ ВБ╝ВъЁ) ьЂ┤въўВіцЖ░ђ ьЋёВџћвАю ьЋўвіћ ВЮўВА┤Вё▒ВЮё ВЎИвХђВЌљВёю ВБ╝ВъЁьЋўвіћ ВёцЖ│ё ьїеьё┤ВЮ┤вІц. ВЃЮВё▒Въљ ВБ╝ВъЁ(Constructor Injection), ВёИьё░ ВБ╝ВъЁ(Setter Injection), ВЮИьё░ьјўВЮ┤Віц ВБ╝ВъЁ ВёИ Ж░ђВДђ в░ЕВІЮВЮ┤ ВъѕвіћвЇ░, ВЃЮВё▒Въљ ВБ╝ВъЁВЮ┤ Ж░ђВъЦ ЖХїВъЦвљювІц. DIPвЦ╝ ВІцьўёьЋўвіћ ЖхгВ▓┤ВаЂВЮИ ЖИ░в▓ЋВЮ┤вІц. FastAPIВЮў `Depends()`Ж░ђ DIВЮў ВІцВаю ЖхгьўёВЮ┤вІц. (JavaВЮў SpringВЮђ @AutowiredвАю, PythonВЮў FastAPIвіћ Depends()вАю, Goвіћ в│┤ьєх ВЃЮВё▒Въљ ьЋеВѕўВЌљВёю DIвЦ╝ ВѕўьќЅьЋювІц.)

    FastAPIВЮў Depends()¶

    FastAPIвЦ╝ ВЊ░Ж│а ВъѕвІцвЕ┤ ВЮ┤в»И DIвЦ╝ ВЊ░Ж│а Въѕвіћ Ж▒░вІц. Depends()Ж░ђ в░ћвАю DI вЕћВ╗цвІѕВдўВЮ┤вІц.

    from fastapi import Depends
from sqlalchemy.orm import Session

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def get_order_repository(db: Session = Depends(get_db)):
    return MySQLOrderRepository(db)

def get_order_service(repo: OrderRepository = Depends(get_order_repository)):
    return OrderService(repo)

@router.post("/orders")
def create_order(
    request: CreateOrderRequest,
    service: OrderService = Depends(get_order_service),
):
    return service.create_order(request)

    create_order ьЋеВѕўвіћ OrderServiceЖ░ђ ьЋёВџћьЋўвІц. FastAPIЖ░ђ ВъљвЈЎВю╝вАю get_order_serviceвЦ╝ ьўИВХюьЋўЖ│а, ЖиИ ВЋѕВЌљВёю get_order_repositoryвЦ╝ ьўИВХюьЋўЖ│а, ЖиИ ВЋѕВЌљВёю get_dbвЦ╝ ьўИВХюьЋювІц. ВЮўВА┤Вё▒ В▓┤ВЮИВЮ┤ ВъљвЈЎВю╝вАю ьЋ┤Ж▓░вљювІц.

    ВЮ┤ ВъљвЈЎ ьЋ┤Ж▓░ВЮё ьЋ┤ВБ╝віћ ВБ╝В▓┤, ВдЅ "вѕёЖ░ђ ВЮўВА┤Вё▒ВЮё ВА░вдйьЋўвіћЖ░ђ"ВЮў ВаюВќ┤Ж░ђ Ж░юв░юВъљВЌљВёю ьћёваѕВъёВЏїьЂгвАю вёўВќ┤Ж░ёвІц. ВЮ┤Ж▓ЃВЮ┤ IoC(Inversion of Control, ВаюВќ┤ВЮў ВЌГВаё)вІц.

    IoC (Inversion of Control, ВаюВќ┤ВЮў ВЌГВаё)¶

    в│┤ьєх ВйћвЊюВЌљВёювіћ Ж░юв░юВъљЖ░ђ ВДЂВаЉ Ж░ЮВ▓┤вЦ╝ вДївЊцЖ│а, ВЌ░Ж▓░ьЋўЖ│а, ВІцьќЅьЋювІц. IoCВЌљВёювіћ ьћёваѕВъёВЏїьЂгЖ░ђ ЖиИ ВЮ╝ВЮё вїђВІа ьЋювІц.

db = SessionLocal()

    Ж░юв░юВъљвіћ "ВЮ┤ ьЋеВѕўвіћ OrderServiceЖ░ђ ьЋёВџћьЋўвІц"Ж│а ВёаВќИвДї ьЋювІц. ВІцВаювАю вДївЊцЖ│а ВЌ░Ж▓░ьЋўвіћ Ж▒┤ FastAPI(ьћёваѕВъёВЏїьЂг)ВЮў вфФВЮ┤вІц.

    "ВаюВќ┤"Ж░ђ "ВЌГВаё"вљљвІц. Ж░юв░юВъљЖ░ђ ьЋўвЇў ВЮ╝(Ж░ЮВ▓┤ ВЃЮВё▒, ВА░вдй)ВЮё ьћёваѕВъёВЏїьЂгЖ░ђ вїђВІаьЋювІц. ВЮ┤Ж▓ї IoCвІц.

    IoC (Inversion of Control, ВаюВќ┤ВЮў ВЌГВаё) ьћёвАюЖиИвъеВЮў ВаюВќ┤ ьЮљвдёВЮё Ж░юв░юВъљЖ░ђ ВЋёвІї ьћёваѕВъёВЏїьЂгЖ░ђ Ж┤ђвдгьЋўвіћ ВёцЖ│ё ВЏљВ╣ЎВЮ┤вІц. Ж░юв░юВъљвіћ "вг┤ВЌЄВЮ┤ ьЋёВџћьЋюЖ░ђ"вЦ╝ ВёаВќИьЋўЖ│а, ьћёваѕВъёВЏїьЂгЖ░ђ "Вќ┤вќ╗Ж▓ї ВА░вдйьЋа Ж▓ЃВЮИЖ░ђ"вЦ╝ Ж▓░ВаЋьЋювІц.
